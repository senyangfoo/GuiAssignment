CREATE TABLE "CUSTOMER"(
"CUST_ID" int not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
"CUST_NAME" VARCHAR(80),
"CUST_EMAIL" VARCHAR(80),
"CUST_PASSWORD" VARCHAR(20)
);

"STAFF_NAME" VARCHAR(80),
"STAFF_EMAIL" VARCHAR(80),

"PROD_NAME" VARCHAR(80),
"CATEGORY" VARCHAR(50),
"DESCRIPTION" VARCHAR(200),
"PRICE" DOUBLE,
"STOCK" INT

CREATE TABLE "ORDERS"(
"ORDER_ID" int not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
"CUST_ID" int not null,
"ORDER_DATE" DATE,
"ADDRESS" VARCHAR(200),
"PHONE_NUM" VARCHAR(50),
"PAYMENTTYPE" VARCHAR(25),
"TOTAL_AMOUNT" DOUBLE,
FOREIGN KEY ("CUST_ID") REFERENCES CUSTOMER("CUST_ID")
);

CREATE TABLE "ORDER_DETAIL" (
"ORDER_PRODUCT_ID" int primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
"ORDER_ID" int not null,
"PROD_ID" int not null,
"QUANTITY" int,
FOREIGN KEY ("ORDER_ID") REFERENCES "ORDERS"("ORDER_ID"),
FOREIGN KEY ("PROD_ID") REFERENCES "PRODUCT"("PROD_ID")
);


CREATE TABLE "CARD_INFO" (
"CARD_INFO_ID" int primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
"CUST_ID" int not null,
"CARD_TYPE" VARCHAR(20),
"CARD_NUMBER" VARCHAR(17),
"EXPIRY_DATE" VARCHAR(10),
"CCV" VARCHAR(4),
FOREIGN KEY ("CUST_ID") REFERENCES "CUSTOMER"("CUST_ID")
);

CREATE TABLE "PRODUCT_REVIEW"(
"PRODUCT_REVIEW_ID" int primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
"CUST_ID" int not null,
"PROD_ID" int not null,
"STAR" int,
"COMMENT" VARCHAR(500),
FOREIGN KEY ("CUST_ID") REFERENCES "CUSTOMER"("CUST_ID"),
FOREIGN KEY ("PROD_ID") REFERENCES "PRODUCT"("PROD_ID")
);
